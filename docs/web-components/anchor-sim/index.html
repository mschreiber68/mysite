Anchor Sim

Component

An Anchor Sim is a generic container component with attributes and a click handler that behave exactly as a normal anchor tag. This can be useful for SEO purposes if you want content to behave as a link without it being possibly seen or indexed by search engines. The Anchor Sim also implements the ARIA "link" role. I had trouble thinking of a good name for this component, but I'm open to ideas!

const KEYCODE = { ENTER: 13, } class AnchorSim extends HTMLElement { constructor() { super(); this.addEventListener('click', this.#onClick.bind(this)); this.addEventListener('keydown', this.#onKeyDown.bind(this)); } connectedCallback() { this.setAttribute('role', 'link'); this.setAttribute('tabindex', '0') } #onClick() { this.#doLinkAction(); } #onKeyDown(event) { switch (event.keyCode) { case KEYCODE.ENTER: this.#doLinkAction(); break; } } #doLinkAction() { const attrs = this .getAttributeNames() .reduce((attrs, attrName) => { attrs[attrName] = this.getAttribute(attrName); return attrs; }, {}); const a = document.createElement('a'); for (const attrName in attrs) { a.setAttribute(attrName, attrs[attrName]); } a.click(); } } window.customElements.define('anchor-sim', AnchorSim);
